## 4. Scoring Methodology (Rule-Based, Weighted)

### **Core Algorithm:**

```typescript
function calculateCaseStrength(session: CaseSession): CaseStrengthResult {
  // 1. Initialize scores
  let totalScore = 0;
  let maxPossibleScore = 0;
  let answeredQuestions = 0;
  
  // 2. Apply weighted scoring
  for (const [questionKey, answer] of Object.entries(session.answers)) {
    const question = getQuestion(questionKey);
    const scoreValue = calculateQuestionScore(question, answer);
    
    totalScore += scoreValue * question.weight;
    maxPossibleScore += question.weight;
    if (answer !== null) answeredQuestions++;
  }
  
  // 3. Normalize to 0-100
  const rawScore = (totalScore / maxPossibleScore) * 100;
  
  // 4. Apply heuristics & overrides
  const adjustedScore = applyHeuristics(rawScore, session);
  
  // 5. Calculate confidence
  const confidence = calculateConfidence(session, answeredQuestions);
  
  // 6. Identify factors
  const strengthFactors = identifyStrengthFactors(session);
  const riskFactors = identifyRiskFactors(session);
  
  // 7. Generate timeline
  const timeline = calculateTimeline(session);
  
  return {
    score: adjustedScore,
    confidence,
    strength_factors: strengthFactors,
    risk_factors: riskFactors,
    timeline,
    // ... other fields
  };
}
```

### **Scoring Weights per Component:**

| Component | Weight | Description |
|-----------|--------|-------------|
| **Evidence Quality** | 40% | Documents, proofs, witnesses available |
| **Procedural Compliance** | 30% | Timelines met, formalities followed |
| **Legal Timeline Status** | 20% | Within limitation, deadlines intact |
| **Documentation Completeness** | 10% | All required docs available |

### **Question Score Calculation:**

```typescript
function calculateQuestionScore(question: Question, answer: any): number {
  switch (question.type) {
    case 'boolean':
      return answer === true ? 1.0 : 0.0;
    
    case 'multiple_choice':
      const option = question.options.find(o => o.value === answer);
      return option?.score_multiplier ?? 0;
    
    case 'date':
      // For dates, check if within valid period
      return isWithinValidPeriod(answer, question) ? 1.0 : 0.0;
    
    case 'amount':
      // Higher amounts might increase jurisdiction options
      return normalizeAmountScore(answer, question);
    
    case 'file':
      // File uploaded = 1.0, not uploaded = 0.0
      return answer !== null ? 1.0 : 0.0;
    
    case 'multi_select':
      // Average of selected options
      const selected = answer as string[];
      const scores = selected.map(val => 
        question.options.find(o => o.value === val)?.score_multiplier ?? 0
      );
      return scores.length > 0 ? average(scores) : 0;
    
    default:
      return 0.5; // Neutral for text fields
  }
}
```

### **Heuristics & Override Rules:**

```typescript
function applyHeuristics(rawScore: number, session: CaseSession): number {
  let adjusted = rawScore;
  
  // Example: Cheque Bounce heuristics
  if (session.notice_type === 'cheque_bounce') {
    // If bank memo exists, minimum score is 40%
    if (session.answers.bank_memo_received === true) {
      adjusted = Math.max(adjusted, 40);
    }
    
    // If demand notice sent late, cap score at 60%
    if (session.answers.demand_notice_sent_within_30_days === false) {
      adjusted = Math.min(adjusted, 60);
    }
    
    // If payment made within 15 days, case is weak
    if (session.answers.drawer_made_payment_within_15_days === true) {
      adjusted = Math.min(adjusted, 30);
    }
  }
  
  // Example: Money Recovery heuristics
  if (session.notice_type === 'money_recovery') {
    // Written agreement significantly boosts score
    if (session.answers.written_agreement === true) {
      adjusted += 10; // Bonus points
    }
    
    // Beyond limitation period = critical risk
    if (session.answers.beyond_limitation === true) {
      adjusted = Math.min(adjusted, 20);
    }
  }
  
  // Ensure score stays in 0-100 range
  return Math.max(0, Math.min(100, adjusted));
}
```

### **Confidence Level Calculation:**

```typescript
function calculateConfidence(
  session: CaseSession,
  answeredQuestions: number
): 'low' | 'medium' | 'high' {
  const totalQuestions = getTotalQuestions(session.notice_type);
  const completionRate = answeredQuestions / totalQuestions;
  const criticalQuestionsAnswered = checkCriticalQuestions(session);
  
  if (completionRate < 0.6 || !criticalQuestionsAnswered) {
    return 'low';
  } else if (completionRate < 0.85) {
    return 'medium';
  } else {
    return 'high';
  }
}
```
